                // checking if an element passed to it is in viewport
                const isInViewport = (el) => {
                    // get the element bounding 
                    let rect = el.getBoundingClientRect(); 
                    
                    // distance from top: rect.top 
                    // distance from left: rect.left 
                    // distance from bottom: rect.bottom 
                    // disance from right: rect.right
                    // inner width of window : window.innerWidth
                    // inner height of window : window.innerHeight
                    // root Elemnt / html element height: document.documentElement.clientHeight
                    // root Elemnt / html element width: document.documentElement.clientWidth
                    
                    return (
                        rect.top >= 0 &&
                        rect.left >= 0 &&
                        rect.bottom <= (window.innerHeight || document.documentElement.clientHeight) && 
                        rect.right <= (window.innerWidth || document.documentElement.clientWidth)
                    );                 
                }
