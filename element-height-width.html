HTML DOM Hints

Determine height and width of an element.

div element with 2px border, 5px marging, 10px padding 
            
                const el = document.querySelector('.test'); 
                const style = window.getComputedStyle(el); 

                // size including padding 
                const clientWidth = el.clientWidth; 
                const clientHeight = el.clientHeight; 

                // size including border and padding 
                const offsetWidth = el.offsetWidth; 
                const offsetHeight = el.offsetHeight; 


                // size excluding padding and border 
                let width = clientWidth - parseFloat(style.paddingLeft) - parseFloat(style.paddingRight); 
                let height = clientHeight - parseFloat(style.paddingTop) - parseFloat(style.paddingBottom); 


                // size with padding, border and margin
                width = offsetWidth + parseFloat(style.marginRight) + parseFloat(style.marginLeft); 
                height = offsetHeight + parseFloat(style.marginTop) + parseFloat(style.marginBottom);
